function matchOption(transcript, options) {

  if (!transcript) return null;

  const t = transcript.toLowerCase().trim();

  for (const o of options) { if (t === o.label.toLowerCase()) return o; }

  for (const o of options) { if (t.includes(o.label.toLowerCase())) return o; }

  for (const o of options) {

    const words = o.label.toLowerCase().split(/\s+/).filter(w=>w.length>2);

    if (words.length>0 && words.every(w=>t.includes(w))) return o;

  }

  for (const o of options) {

    const first = o.label.toLowerCase().split(/\s+/).find(w=>w.length>3);

    if (first && t.includes(first)) return o;

  }

  return null;

}


function extractNumber(transcript, min=0, max=10) {

  if (!transcript) return null;

  const t = transcript.toLowerCase().trim();

  for (const [phrase, val] of Object.entries(WORD_TO_NUM)) {

    if (t.includes(phrase)) {

      const scaled = max===100 && val<=10 ? val*10 : val;

      if (scaled>=min && scaled<=max) return scaled;

    }

  }

  const nums = t.match(/\d+/g);

  if (nums) { const n=parseInt(nums[0]); if (n>=min && n<=max) return n; }

  return null;

}

function VoiceProvider({children}) {

  const [voiceMode, setVoiceMode] = useState(false);

  const [listening, setListening] = useState(false);

  const [speaking, setSpeaking] = useState(false);

  const [transcript, setTranscript] = useState('');

  const [lastMatch, setLastMatch] = useState('');

  const [supported, setSupported] = useState({tts:false,stt:false});

  const recogRef = useRef(null);

  const synthRef = useRef(null);


  useEffect(()=>{

    const hasTTS = 'speechSynthesis' in window;

    const SR = window.SpeechRecognition || window.webkitSpeechRecognition;

    setSupported({tts:hasTTS, stt:!!SR});

    if (hasTTS) synthRef.current = window.speechSynthesis;

    if (SR) {

      const r = new SR();

      r.lang='en-GB'; r.continuous=false; r.interimResults=true; r.maxAlternatives=3;

      recogRef.current = r;

    }

  },[]);


  const speak = useCallback((text, onEnd)=>{

    if (!synthRef.current) return;

    synthRef.current.cancel();

    if (!text) { if(onEnd) onEnd(); return; }

    const utt = new SpeechSynthesisUtterance(text);

    utt.lang='en-GB'; utt.rate=0.88; utt.pitch=1; utt.volume=1;

    utt.onstart=()=>setSpeaking(true);

    utt.onend=()=>{ setSpeaking(false); if(onEnd) onEnd(); };

    utt.onerror=()=>{ setSpeaking(false); if(onEnd) onEnd(); };

    synthRef.current.speak(utt);

  },[]);


  const stopSpeaking = useCallback(()=>{

    if(synthRef.current) synthRef.current.cancel();

    setSpeaking(false);

  },[]);


  const listen = useCallback((onFinal, onInterim)=>{

    const r = recogRef.current;

    if (!r) return;

    try { r.stop(); } catch(e) {}

    setTranscript(''); setLastMatch(''); setListening(true);

    r.onresult=(e)=>{

      let interim='', final='';

      for (let i=e.resultIndex; i<e.results.length; i++) {

        if (e.results[i].isFinal) final+=e.results[i][0].transcript;

        else interim+=e.results[i][0].transcript;

      }

      const cur = final||interim;

      setTranscript(cur);

      if (onInterim) onInterim(cur);

      if (final) { setListening(false); if(onFinal) onFinal(final.trim()); }

    };

    r.onerror=()=>setListening(false);

    r.onend=()=>setListening(false);

    try { r.start(); } catch(e) { setListening(false); }

  },[]);


  const stopListening = useCallback(()=>{

    try { recogRef.current?.stop(); } catch(e) {}

    setListening(false);

  },[]);


  return (

    <VoiceCtx.Provider value={{voiceMode,setVoiceMode,listening,speaking,transcript,lastMatch,setLastMatch,supported,speak,stopSpeaking,listen,stopListening}}>

      {children}

    </VoiceCtx.Provider>

  );

}

function calcScores(d) {

  const int=(v,def=0)=>parseInt(v??def)||0;

  const sym=int(d.b1_walk)+int(d.b1_stairs)+int(d.b1_normal);

  const sat=['b2_sit','b2_lie','b2_getup','b2_house','b2_rec'].reduce((s,k)=>s+int(d[k]),0);

  const exp=['b3_pain','b3_func','b3_overall'].reduce((s,k)=>s+int(d[k]),0);

  const wlk=int(d.b4_aids)+int(d.b4_stand)+int(d.b4_dist);

  const std=['b4s_up','b4s_dn','b4s_kneel','b4s_rise','b4s_floor','b4s_lat'].reduce((s,k)=>s+int(d[k]),0);

  const adv=['b4a_twist','b4a_floor','b4a_run','b4a_jump','b4a_stop'].reduce((s,k)=>s+int(d[k]),0);

  const sel=d.disc_sel||[];

  const disc=sel.slice(0,3).reduce((s,a)=>s+int(d['disc_'+a]),0);

  const func=wlk+std+adv+disc;

  const subjective=sym+sat+exp;

  const vas=int(d.vas);

  const vas_preop=d.vas_preop!==undefined?int(d.vas_preop):null;

  const vas_delta=vas_preop!==null?vas-vas_preop:null;

  const lefRaw=['lef1','lef2','lef3','lef4'].reduce((s,k)=>s+int(d[k]),0);

  const piRaw=int(d.pi1)+int(d.pi2)+int(d.pi3);

  const pifRaw=['pif1','pif2','pif3','pif4'].reduce((s,k)=>s+int(d[k]),0);

  const eq5dSum=['eq_mob','eq_self','eq_usual','eq_pain','eq_anx'].reduce((s,k)=>s+int(d[k]),0);

  const eqVas=int(d.eq_vas,50);

  return {sym,sat,exp,wlk,std,adv,disc,func,subjective,vas,vas_preop,vas_delta,lefRaw,piRaw,pifRaw,eq5dSum,eqVas};

}


/* ═══════════════════════ GOOGLE SHEETS WEBHOOK ═══════════════════════ */

async function sendToSheets(payload) {

  if (!WEBHOOK_URL) return { ok: false, reason: 'no_url' };

  try {

    // Google Apps Script requires no-cors; response cannot be read but data is written

    await fetch(WEBHOOK_URL, {

      method: 'POST',

      mode: 'no-cors',

      headers: { 'Content-Type': 'application/json' },

      body: JSON.stringify(payload),

    });

    return { ok: true };

  } catch (e) {

    console.warn('Sheets webhook failed:', e);

    return { ok: false, reason: e.message };

  }

}

  const [sheetsOk, setSheetsOk] = useState(false);

  const go = useCallback((n) => {

    setStep(n);

    window.scrollTo({top:0,behavior:'smooth'});

    // Save partial on every step advance (drop-off tracking)

    if (n > 0 && n < 18) {

      const scores = calcScores(data);

      setSubs(prev => {

        const rec = {id:token,token,timestamp:new Date().toISOString(),lastStep:n,status:'partial',data:{...data},scores};

        const existing = prev.findIndex(s=>s.token===token);

        if (existing >= 0) { const u=[...prev]; u[existing]=rec; return u; }

        return [...prev,rec];

      });

      // Send partial with contact info to Sheets if contact provided

      if (data.contact_email || data.contact_phone) {

        sendToSheets({...buildPayload(token,'partial',n,data,calcScores(data)), record_type:'partial'});

      }

    }

  }, [data, token]);

    setSubmitting(true);

    const scores = calcScores(data);
